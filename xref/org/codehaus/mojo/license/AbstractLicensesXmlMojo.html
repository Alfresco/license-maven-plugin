44  /**
45   * A common parent for {@link LicensesXmlInsertVersionsMojo} and {@link AbstractDownloadLicensesMojo}.
46   *
47   * @author <a href="https://github.com/ppalaga">Peter Palaga</a>
48   * @since 1.19
49   */
50  public abstract class AbstractLicensesXmlMojo
51      extends AbstractMojo
52  {
53  
54      /**
55       * The output file containing a mapping between each dependency and it's license information.
56       *
57       * @since 1.0
58       */
59      @Parameter( property = "licensesOutputFile",
60                      defaultValue = "${project.build.directory}/generated-resources/licenses.xml" )
61      protected File licensesOutputFile;
62  
63      /**
64       * An end of line constant name denoting the EOL string to use when redering the {@code licenses.xml} file. Possible
65       * values are {@code LF}, {@code CRLF}, {@code AUTODETECT} and {@code PLATFORM}.
66       * <p>
67       * When the value {@code AUTODETECT} is used, the mojo will use whatever EOL value is used in the first existing of
68       * the following files: {@link #licensesConfigFile}, <code>${basedir}/pom.xml</code>.
69       * <p>
70       * The value {@code PLATFORM} is deprecated but still kept for backwards compatibility reasons.
71       *
72       * @since 1.17
73       */
74      @Parameter( property = "licensesOutputFileEol", defaultValue = "AUTODETECT" )
75      protected Eol licensesOutputFileEol;
76  
77      /**
78       * Encoding used to (1) read the file specified in {@link #licensesConfigFile} and (2) write the file specified in
79       * {@link #licensesOutputFile}.
80       *
81       * @since 1.17
82       */
83      @Parameter( property = "licensesOutputFileEncoding", defaultValue = "${project.build.sourceEncoding}" )
84      private String licensesOutputFileEncoding;
85  
86      /**
87       * Location of the local repository.
88       *
89       * @since 1.0
90       */
91      @Parameter( defaultValue = "${localRepository}", readonly = true )
92      protected ArtifactRepository localRepository;
93  
94      /**
95       * List of Remote Repositories used by the resolver
96       *
97       * @since 1.0
98       */
99      @Parameter( defaultValue = "${project.remoteArtifactRepositories}", readonly = true )
100     protected List<ArtifactRepository> remoteRepositories;
101 
102     /**
103      * The Maven Project Object
104      *
105      * @since 1.0
106      */
107     @Parameter( defaultValue = "${project}", readonly = true )
108     protected MavenProject project;
109 
110     /**
111      * Dependencies tool.
112      *
113      * @since 1.0
114      */
115     @Component
116     protected LicensedArtifactResolver dependenciesTool;
117 
118     private Charset charset;
119 
120     /** {@inheritDoc} */
121     public String getEncoding()
122     {
123         initEncoding();
124         return licensesOutputFileEncoding;
125     }
126 
127     Charset getCharset()
128     {
129         initEncoding();
130         return charset;
131     }
132 
133     private void initEncoding()
134     {
135         if ( charset == null )
136         {
137             if ( licensesOutputFileEncoding == null )
138             {
139                 licensesOutputFileEncoding = System.getProperty( "file.encoding" );
140                 getLog().warn( "Using the default system encoding for reading or writing licenses.xml file."
141                     + " This makes your build platform dependent. You should set either"
142                     + " project.build.sourceEncoding or licensesOutputFileEncoding" );
143             }
144             charset = Charset.forName( licensesOutputFileEncoding );
145 
146             if ( licensesOutputFileEol == Eol.AUTODETECT )
147             {
148                 final Path[] paths = getAutodetectEolFiles();
149                 Path autodetectFromFile = null;
150                 for ( Path path : paths )
151                 {
152                     if ( Files.exists( path ) )
153                     {
154                         autodetectFromFile = path;
155                         break;
156                     }
157                 }
158                 if ( autodetectFromFile != null )
159                 {
160                     try
161                     {
162                         licensesOutputFileEol = Eol.autodetect( autodetectFromFile, charset );
163                     }
164                     catch ( IOException e )
165                     {
166                         throw new RuntimeException( "Cannot autodetect end of line from file \"" + autodetectFromFile
167                             + "\"", e );
168                     }
169                 }
170             }
171         }
172     }
173 
174     protected Path[] getAutodetectEolFiles()
175     {
176         return new Path[] { licensesOutputFile.toPath() };
177     }
178 
179     protected void writeLicenseSummary( List<ProjectLicenseInfo> deps, File licensesOutputFile, boolean writeVersions )
180         throws ParserConfigurationException, TransformerException, IOException
181     {
182         initEncoding();
183         LicenseSummaryWriter.writeLicenseSummary( deps, licensesOutputFile, charset, licensesOutputFileEol,
184                                                   writeVersions );
185     }
186 
187 }
