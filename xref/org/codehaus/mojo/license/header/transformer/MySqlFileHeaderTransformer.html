3   import org.codehaus.plexus.component.annotations.Component;

5   /**
6    * Implementation of {@link FileHeaderTransformer} for mysql format.
7    *
8    * <strong>Note:</strong> Using mysql files, you can not just use the {@code sql} file header transformer
9    * (see http://jira.codehaus.org/browse/MLICENSE-56) for more informations about the problem.
10   *
11   * @author tchemit dev@tchemit.fr
12   * @since 1.4
13   */
14  @Component( role = FileHeaderTransformer.class, hint = "mysql" )
15  public class MySqlFileHeaderTransformer
16      extends AbstractFileHeaderTransformer
17  {
18  
19      /**
20       * Default constructor.
21       */
22      public MySqlFileHeaderTransformer()
23      {
24          super( "mysql", "header transformer with mysql comment style", "-- -", "-- -", "-- " );
25      }
26  
27      /**
28       * {@inheritDoc}
29       */
30      public String[] getDefaultAcceptedExtensions()
31      {
32          return new String[]{ "mysql" };
33      }
34  }
