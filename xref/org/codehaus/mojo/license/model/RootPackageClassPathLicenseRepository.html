5   /**
6    * A special class-path license repository defined on root java package.
7    * Created on 21/05/16.
8    *
9    * @author Tony Chemit - dev@tchemit.fr
10   * @since 1.9
11   */
12  public class RootPackageClassPathLicenseRepository
13      extends LicenseRepository
14  {
15  
16      @Override
17      protected URL getDefinitionURL()
18      {
19          return getClass().getClassLoader().getResource( REPOSITORY_DEFINITION_FILE );
20      }
21  
22      @Override
23      protected URL getLicenseBaseURL( String licenseName )
24      {
25          return getClass().getClassLoader().getResource( licenseName );
26      }
27  
28  }
