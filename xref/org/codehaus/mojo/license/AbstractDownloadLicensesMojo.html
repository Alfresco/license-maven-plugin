94      /**
95       * A file containing the license data (most notably license names and license URLs) missing in
96       * {@code pom.xml} files of the dependencies.
97       * <p>
98       * Note that since 1.18, if you set {@link #errorRemedy} to {@code xmlOutput} the format of
99       * {@link #licensesErrorsFile} is the same as the one of {@link #licensesConfigFile}. So you can use
100      * {@link #licensesErrorsFile} as a base for {@link #licensesConfigFile}.
101      * <p>
102      * Since 1.18, the format of the file is as follows:
103      * <pre>
104      * {@code
105      * <licenseSummary>
106      *   <dependencies>
107      *     <dependency>
108      *       <groupId>\Qaopalliance\E</groupId><!-- A regular expression -->
109      *       <artifactId>\Qaopalliance\E</artifactId><!-- A regular expression -->
110      *       <!-- <version>.*</version> A version pattern is optional, .* being the default.
111      *       <matchLicenses>
112      *         <!-- Match a list of licenses with a single entry having name "Public Domain" -->
113      *         <license>
114      *           <name>\QPublic Domain\E</name><!-- A regular expression -->
115      *         </license>
116      *       </matchLicenses>
117      *       <licenses approved="true" /><!-- Leave the matched dependency as is. -->
118      *                                   <!-- In this particular case we approve that code in the Public -->
119      *                                   <!-- Domain does not need any license URL. -->
120      *     </dependency>
121      *     <dependency>
122      *         <groupId>\Qasm\E</groupId>
123      *       <artifactId>\Qasm\E</artifactId>
124      *       <matchLicenses>
125      *         <!-- Match an empty list of licenses -->
126      *       </matchLicenses>
127      *       <!-- Replace the list of licenses in all matching dependencies with the following licenses -->
128      *       <licenses>
129      *         <license>
130      *           <name>BSD 3-Clause ASM</name>
131      *           <url>https://gitlab.ow2.org/asm/asm/raw/ASM_3_1_MVN/LICENSE.txt</url>
132      *         </license>
133      *       </licenses>
134      *     </dependency>
135      *     <dependency>
136      *       <groupId>\Qca.uhn.hapi\E</groupId>
137      *       <artifactId>.*</artifactId>
138      *       <matchLicenses>
139      *         <!-- Match a list of licenses with the following three entries in order: -->
140      *         <license>
141      *           <name>\QHAPI is dual licensed (MPL, GPL)\E</name>
142      *           <comments>\QHAPI is dual licensed under both the Mozilla Public License and the GNU General Public License.\E\s+\QWhat this means is that you may choose to use HAPI under the terms of either license.\E\s+\QYou are both permitted and encouraged to use HAPI, royalty-free, within your applications,\E\s+\Qwhether they are free/open-source or commercial/closed-source, provided you abide by the\E\s+\Qterms of one of the licenses below.\E\s+\QYou are under no obligations to inform the HAPI project about what you are doing with\E\s+\QHAPI, but we would love to hear about it anyway!\E</comments>
143      *         </license>
144      *         <license>
145      *           <name>\QMozilla Public License 1.1\E</name>
146      *           <url>\Qhttp://www.mozilla.org/MPL/MPL-1.1.txt\E</url>
147      *           <file>\Qmozilla public license 1.1 - index.0c5913925d40.txt\E</file>
148      *         </license>
149      *         <license>
150      *           <name>\QGNU General Public License\E</name>
151      *           <url>\Qhttp://www.gnu.org/licenses/gpl.txt\E</url>
152      *           <file>\Qgnu general public license - gpl.txt\E</file>
153      *         </license>
154      *       </matchLicenses>
155      *       <licenses approved="true" /><!-- It is OK that the first entry has no URL -->
156      *     </dependency>
157      *   </dependencies>
158      * </licenseSummary>
159      * }
160      * </pre>
161      * <p>
162      * Before 1.18 the format was the same as the one of {@link #licensesOutputFile} and the groupIds and artifactIds
163      * were matched as plain text rather than regular expressions. No other elements (incl. versions) were matched at
164      * all. Since 1.18 the backwards compatibility is achieved by falling back to plain text matching of groupIds
165      * and artifactIds if the given {@code <dependency>} does not contain the {@code <matchLicenses>} element.
166      * <p>
167      * Relationship to other parameters:
168      * <ul>
169      * <li>License names and license URLs {@link #licensesConfigFile} is applied before
170      * {@link #licenseUrlReplacements}</li>
171      * <li>{@link #licenseUrlReplacements} are applied before {@link #licenseUrlFileNames}</li>
172      * <li>{@link #licenseUrlFileNames} have higher precedence than {@code <file>} elements in
173      * {@link #licensesConfigFile}</li>
174      * <li>{@link #licenseUrlFileNames} are ignored when {@link #organizeLicensesByDependencies} is {@code true}</li>
175      * </ul>
176      *
177      * @since 1.0
178      */
179     @Parameter( property = "licensesConfigFile", defaultValue = "${project.basedir}/src/license/licenses.xml" )
180     protected File licensesConfigFile;

434     /**
435      * List of regexps/replacements applied to the license urls prior to download.
436      *
437      * <p>License urls that match a regular expression will be replaced by the corresponding
438      * replacement. Replacement is performed with {@link java.util.regex.Matcher#replaceAll(String)
439      * java.util.regex.Matcher#replaceAll(String)} so you can take advantage of
440      * capturing groups to facilitate flexible transformations.</p>
441      *
442      * <p>If the replacement element is omitted, this is equivalent to an empty replacement string.</p>
443      *
444      * <p>The replacements are applied in the same order as they are present in the configuration. The default
445      * replacements (that can be activated via {@link #useDefaultUrlReplacements}) are appended to
446      * {@link #licenseUrlReplacements}</p>
447      *
448      * <p>The {@code id} field of {@link LicenseUrlReplacement} is optional and is useful only if you want to override
449      * some of the default replacements.</p>
450      *
451      * <pre>
452      * {@code
453      *
454      * <licenseUrlReplacements>
455      *   <licenseUrlReplacement>
456      *     <regexp>\Qhttps://glassfish.java.net/public/CDDL+GPL_1_1.html\E</regexp>
457      *     <replacement>https://oss.oracle.com/licenses/CDDL+GPL-1.1</replacement>
458      *   </licenseUrlReplacement>
459      *   <licenseUrlReplacement>
460      *     <regexp>https://(.*)</regexp><!-- replace https with http -->
461      *     <replacement>http://$1</replacement>;
462      *   </licenseUrlReplacement>
463      *   <licenseUrlReplacement>
464      *     <id>github.com-0</id><!-- An optional id to override the default replacement with the same id -->
465      *     <regexp>^https?://github\.com/([^/]+)/([^/]+)/blob/(.*)$</regexp><!-- replace GitHub web UI with raw -->
466      *     <replacement>https://raw.githubusercontent.com/$1/$2/$3</replacement>
467      *   </licenseUrlReplacement>
468      * </licenseUrlReplacements>
469      * }
470      * </pre>
471      * <p>
472      * Relationship to other parameters:
473      * <ul>
474      * <li>Default URL replacements can be unlocked by setting {@link #useDefaultUrlReplacements} to {@code true}.</li>
475      * <li>License names and license URLs {@link #licensesConfigFile} is applied before
476      * {@link #licenseUrlReplacements}</li>
477      * <li>{@link #licenseUrlReplacements} are applied before {@link #licenseUrlFileNames}</li>
478      * <li>{@link #licenseUrlFileNames} have higher precedence than {@code <file>} elements in
479      * {@link #licensesConfigFile}</li>
480      * <li>{@link #licenseUrlFileNames} are ignored when {@link #organizeLicensesByDependencies} is {@code true}</li>
481      * </ul>
482      *
483      * @since 1.17
484      */

507     /**
508      * A map that helps to select local files names for the content downloaded from license URLs.
509      * <p>
510      * Keys in the map are the local file names. These files will be created under {@link #licensesOutputDirectory}.
511      * <p>
512      * Values are white space ({@code " \t\n\r"}) separated lists of regular expressions that will be used to match
513      * license URLs. The regular expressions are compiled using {@link Pattern#CASE_INSENSITIVE}. Note that various
514      * characters that commonly occur in URLs have special meanings in regular extensions. Therefore, consider using
515      * regex quoting as described in {@link Pattern} - e.g. {@code http://example\.com} or
516      * {@code \Qhttp://example.com\E}
517      * <p>
518      * In addition to URL patterns, the list can optionally contain a sha1 checksum of the expected content. This is to
519      * ensure that the content delivered by a URL does not change without notice. Note that strict checking
520      * of the checksum happens only when {@link #forceDownload} is {@code true}. Otherwise the mojo assumes that the URL
521      * -&gt; local name mapping is correct and downloads from the URL only if the local file does not exist.
522      * <p>
523      * A special value-less entry {@code <spdx/>} can be used to activate built-in license names that are based on
524      * license IDs from <a href="https://spdx.org/licenses/">https://spdx.org/licenses</a>. The built-in SPDX mappings
525      * can be overridden by the subsequent entries. To see which SPDX mappings are built-in, add the {@code <spdx/>}
526      * entry and run the mojo with debug log level, e.g. using {@code -X} or
527      * {-Dorg.slf4j.simpleLogger.log.org.codehaus.mojo.license=debug} on the command line.
528      * <p>
529      * An example:
530      * <pre>
531      * {@code
532      * <licenseUrlFileNames>
533      *   <spdx/><!-- A special element to activate built-in file name entries based on spdx.org license IDs -->
534      *   <bsd-antlr.html>
535      *       sha1:81ffbd1712afe8cdf138b570c0fc9934742c33c1
536      *       https?://(www\.)?antlr\.org/license\.html
537      *   </bsd-antlr.html>
538      *   <cddl-gplv2-ce.txt>
539      *       sha1:534a3fc9ae1076409bb00d01127dbba1e2620e92
540      *       \Qhttps://raw.githubusercontent.com/javaee/activation/master/LICENSE.txt\E
541      *   </cddl-gplv2-ce.txt>
542      * </licenseUrlFileNames>
543      * }
544      * </pre>
545      * <p>
546      * Relationship to other parameters:
547      * <ul>
548      * <li>License names and license URLs {@link #licensesConfigFile} is applied before
549      * {@link #licenseUrlReplacements}</li>
550      * <li>{@link #licenseUrlReplacements} are applied before {@link #licenseUrlFileNames}</li>
551      * <li>{@link #licenseUrlFileNames} have higher precedence than {@code <file>} elements in
552      * {@link #licensesConfigFile}</li>
553      * <li>{@link #licenseUrlFileNames} are ignored when {@link #organizeLicensesByDependencies} is {@code true}</li>
554      * </ul>
555      *
556      * @since 1.18
557      */
558     @Parameter
559     protected Map<String, String> licenseUrlFileNames;

561     /**
562      * A list of regexp:replacement pairs that should be applied to file names for storing licenses.
563      * <p>

698     /**
699      * {@inheritDoc}
700      * @throws MojoFailureException
701      */
702     public void execute()
703         throws MojoExecutionException, MojoFailureException
704     {
705 
706         if ( isSkip() )
707         {
708             LOG.info( "skip flag is on, will skip goal." );
709             return;
710         }
711 
712         this.errorRemedy = getEffectiveErrorRemedy( this.quiet, this.errorRemedy );
713         this.preferredFileNames = PreferredFileNames.build( licensesOutputDirectory, licenseUrlFileNames );
714         this.cache = new Cache( licenseUrlFileNames != null && !licenseUrlFileNames.isEmpty() );
715         this.urlReplacements = urlReplacements();
716 
717         initDirectories();
718 
719         final LicenseMatchers matchers = LicenseMatchers.load( licensesConfigFile );
720 
721         if ( !forceDownload )
722         {
723             try
724             {
725                 final List<ProjectLicenseInfo> projectLicenseInfos =
726                                 LicenseSummaryReader.parseLicenseSummary( licensesOutputFile );
727                 for ( ProjectLicenseInfo dep : projectLicenseInfos )
728                 {
729                     for ( ProjectLicense lic : dep.getLicenses() )
730                     {
731                         final String fileName = lic.getFile();
732                         if ( fileName != null )
733                         {
734                             orphanFileNames.add( fileName );
735                             final String url = lic.getUrl();
736                             if ( url != null )
737                             {
738                                 final File file = new File( licensesOutputDirectory, fileName );
739                                 if ( file.exists() )
740                                 {
741                                     final LicenseDownloadResult entry =
742                                         LicenseDownloadResult.success( file, FileUtil.sha1( file.toPath() ), false );
743                                     cache.put( url, entry );
744                                 }
745                             }
746                         }
747                     }
748                 }
749             }
750             catch ( Exception e )
751             {
752                 throw new MojoExecutionException( "Unable to process license summary file: " + licensesOutputFile, e );
753             }
754         }
755 
756         final Map<String, LicensedArtifact> dependencies = getDependencies();
757 
758         // The resulting list of licenses after dependency resolution
759         final List<ProjectLicenseInfo> depProjectLicenses = new ArrayList<>();
760 
761         try ( LicenseDownloader licenseDownloader =
762             new LicenseDownloader( findActiveProxy(), connectTimeout, socketTimeout, connectionRequestTimeout,
763                                    contentSanitizers(), getCharset() ) )
764         {
765             for ( LicensedArtifact artifact : dependencies.values() )
766             {
767                 LOG.debug( "Checking licenses for project " + artifact );
768                 final ProjectLicenseInfo depProject = createDependencyProject( artifact );
769                 matchers.replaceMatches( depProject );
770 
771                 /* Copy the messages and handle them via handleError() that may eventually add them back */
772                 final List<String> msgs = new ArrayList<>( depProject.getDownloaderMessages() );
773                 depProject.getDownloaderMessages().clear();
774                 for ( String msg : msgs )
775                 {
776                     handleError( depProject, msg );
777                 }
778 
779                 depProjectLicenses.add( depProject );
780             }
781             if ( !offline )
782             {
783                 /* First save the matching URLs into the cache */
784                 for ( ProjectLicenseInfo depProject : depProjectLicenses )
785                 {
786                     downloadLicenses( licenseDownloader, depProject, true );
787                 }
788                 LOG.debug( "Finished populating cache" );
789                 /*
790                  * Then attempt to download the rest of the URLs using the available cache entries to select local
791                  * file names based on file content sha1
792                  */
793                 for ( ProjectLicenseInfo depProject : depProjectLicenses )
794                 {
795                     downloadLicenses( licenseDownloader, depProject, false );
796                 }
797             }
798         }
799         catch ( IOException e )
800         {
801             throw new RuntimeException( e );
802         }
803 
804         try
805         {
806             if ( sortByGroupIdAndArtifactId )
807             {
808                 sortByGroupIdAndArtifactId( depProjectLicenses );
809             }
810 
811             List<ProjectLicenseInfo> depProjectLicensesWithErrors = filterErrors( depProjectLicenses );
812             writeLicenseSummary( depProjectLicenses, licensesOutputFile, writeVersions );
813             if ( depProjectLicensesWithErrors != null && !depProjectLicensesWithErrors.isEmpty() )
814             {
815                 writeLicenseSummary( depProjectLicensesWithErrors, licensesErrorsFile, writeVersions );
816             }
817 
818             removeOrphanFiles( depProjectLicenses );
819         }
820         catch ( Exception e )
821         {
822             throw new MojoExecutionException( "Unable to write license summary file: " + licensesOutputFile, e );
823         }
824 
825         switch ( errorRemedy )
826         {
827             case ignore:
828             case failFast:
829                 /* do nothing */
830                 break;
831             case warn:
832                 LOG.warn( "There were {} download errors - check the warnings above", downloadErrorCount );
833                 break;
834             case xmlOutput:
835                 if ( downloadErrorCount > 0 )
836                 {
837                     throw new MojoFailureException( "There were " + downloadErrorCount + " download errors - check "
838                         + licensesErrorsFile.getAbsolutePath() );
839                 }
840                 break;
841             default:
842                 throw new IllegalStateException( "Unexpected value of " + ErrorRemedy.class.getName() + ": "
843                     + errorRemedy );
844         }
845     }

850         if ( licenseUrlReplacements != null )

852             for ( LicenseUrlReplacement r : licenseUrlReplacements )
853             {
854                 b.replacement( r.getId(), r.getRegexp(), r.getReplacement() );
855             }
856         }
857         return b.build();
858     }

860     private Map<String, ContentSanitizer> contentSanitizers()
861     {
862         Map<String, ContentSanitizer> result =
863             new TreeMap<String, ContentSanitizer>();
864         if ( useDefaultContentSanitizers )
865         {
866             final Map<String, ContentSanitizer> defaultSanitizers =
867                 SpdxLicenseList.getLatest().getAttachments().getContentSanitizers();
868             result.putAll( defaultSanitizers  );
869             if ( LOG.isDebugEnabled() && !defaultSanitizers.isEmpty() )
870             {
871                 final StringBuilder sb = new StringBuilder() //
872                         .append( "Applied " ) //
873                         .append( defaultSanitizers.size() ) //
874                         .append( " licenseContentSanitizers:\n<licenseContentSanitizers>\n" );
875                 for ( ContentSanitizer sanitizer : defaultSanitizers.values() )
876                 {
877                     sb.append( "  <licenseContentSanitizer>\n" ) //
878                       .append( "    <id>" ) //
879                       .append( sanitizer.getId() ) //
880                       .append( "</id>\n" ) //
881                       .append( "    <urlRegexp>" ) //
882                       .append( StringEscapeUtils.escapeJava( sanitizer.getUrlPattern().pattern() ) ) //
883                       .append( "</urlRegexp>\n" ) //
884                       .append( "    <contentRegexp>" ) //
885                       .append( StringEscapeUtils.escapeJava( sanitizer.getContentPattern().pattern() ) ) //
886                       .append( "</contentRegexp>\n" ) //
887                       .append( "    <contentReplacement>" ) //
888                       .append( StringEscapeUtils.escapeJava( sanitizer.getContentReplacement() ) ) //
889                       .append( "</contentReplacement>\n" ) //
890                       .append( "  </licenseContentSanitizer>\n" );
891                 }
892                 sb.append( "</licenseContentSanitizers>" );
893 
894                 LOG.debug( sb.toString() );
895             }
896         }
897         if ( licenseContentSanitizers != null )
898         {
899             for ( LicenseContentSanitizer s : licenseContentSanitizers )
900             {
901                 result.put( s.getId(),
902                             ContentSanitizer.compile( s.getId(), s.getUrlRegexp(),
903                                                                                   s.getContentRegexp(),
904                                                                                   s.getContentReplacement() ) );
905             }
906         }
907 
908         return Collections.unmodifiableMap( result );
909     }

935     /**
936      * Removes from the given {@code depProjectLicenses} those elements which have non-empty
937      * {@link ProjectLicenseInfo#getDownloaderMessages()} and adds those to the resulting {@link List}.
938      *
939      * @param depProjectLicenses the list of {@link ProjectLicenseInfo}s to filter
940      * @return a new {@link List} of {@link ProjectLicenseInfo}s containing only elements with non-empty
941      *         {@link ProjectLicenseInfo#getDownloaderMessages()}
942      */
943     private List<ProjectLicenseInfo> filterErrors( List<ProjectLicenseInfo> depProjectLicenses )
944     {
945         final List<ProjectLicenseInfo> result = new ArrayList<>();
946         final Iterator<ProjectLicenseInfo> it = depProjectLicenses.iterator();
947         while ( it.hasNext() )
948         {
949             final ProjectLicenseInfo dep = it.next();
950             final List<String> messages = dep.getDownloaderMessages();
951             if ( messages != null && !messages.isEmpty() )
952             {
953                 it.remove();
954                 result.add( dep );
955             }
956         }
957         return result;
958     }

971     private void sortByGroupIdAndArtifactId( List<ProjectLicenseInfo> depProjectLicenses )
972     {
973         Comparator<ProjectLicenseInfo> comparator = new Comparator<ProjectLicenseInfo>()
974         {
975             public int compare( ProjectLicenseInfo info1, ProjectLicenseInfo info2 )
976             {
977                 //ProjectLicenseInfo::getId() can not be used because . is before : thus a:b.c would be after a.b:c
978                 return ( info1.getGroupId() + "+" + info1.getArtifactId() ).compareTo( info2.getGroupId()
979                         + "+" + info2.getArtifactId() );
980             }
981         };
982         Collections.sort( depProjectLicenses, comparator );
983     }

1104     /**
1105      * Determine filename to use for downloaded license file. The file name is based on the configured name of the
1106      * license (if available) and the remote filename of the license.
1107      *
1108      * @param depProject the project containing the license
1109      * @param licenseUrl the license url
1110      * @param licenseName the license name
1111      * @param string
1112      * @return A filename to be used for the downloaded license file
1113      * @throws URISyntaxException
1114      */
1115     private FileNameEntry getLicenseFileName( ProjectLicenseInfo depProject, final String url,
1116                                                            final String licenseName, String licenseFileName )
1117         throws URISyntaxException
1118     {
1119 
1120         final URI licenseUrl = new URI( url );
1121         File licenseUrlFile = new File( licenseUrl.getPath() );
1122 
1123         if ( organizeLicensesByDependencies )
1124         {
1125             if ( licenseFileName != null && !licenseFileName.isEmpty() )
1126             {
1127                 return new FileNameEntry( new File( licensesOutputDirectory, new File( licenseFileName ).getName() ),
1128                                           false, null );
1129             }
1130             licenseFileName = String.format( "%s.%s%s", depProject.getGroupId(), depProject.getArtifactId(),
1131                                              licenseName != null
1132                                                  ? "_" + licenseName
1133                                                  : "" ).replaceAll( "\\s+", "_" );
1134         }
1135         else
1136         {
1137             final FileNameEntry preferredFileNameEntry = preferredFileNames.getEntryByUrl( url );
1138             if ( preferredFileNameEntry != null )
1139             {
1140                 return preferredFileNameEntry;
1141             }
1142 
1143             if ( licenseFileName != null && !licenseFileName.isEmpty() )
1144             {
1145                 return new FileNameEntry( new File( licensesOutputDirectory,
1146                                                                          new File( licenseFileName ).getName() ),
1147                                                                false, null );
1148             }
1149 
1150             licenseFileName = licenseUrlFile.getName();
1151 
1152             if ( licenseName != null )
1153             {
1154                 licenseFileName = licenseName + " - " + licenseUrlFile.getName();
1155             }
1156 
1157             // Normalize whitespace
1158             licenseFileName = licenseFileName.replaceAll( "\\s+", " " );
1159         }
1160 
1161         // lower case and (back)slash removal
1162         licenseFileName = licenseFileName.toLowerCase( Locale.US ).replaceAll( "[\\\\/]+", "_" );
1163 
1164         licenseFileName = sanitize( licenseFileName );
1165 
1166         return new FileNameEntry( new File( licensesOutputDirectory, licenseFileName ), false, null );
1167     }
1168 
1169     private String sanitize( String licenseFileName )
1170     {
1171         if ( licenseUrlFileNameSanitizers != null )
1172         {
1173             for ( LicenseUrlReplacement sanitizer : licenseUrlFileNameSanitizers )
1174             {
1175                 Pattern regexp = sanitizer.getPattern();
1176                 String replacement = sanitizer.getReplacement() == null ? "" : sanitizer.getReplacement();
1177                 if ( regexp != null )
1178                 {
1179                     licenseFileName = regexp.matcher( licenseFileName ).replaceAll( replacement );
1180                 }
1181             }
1182         }
1183         return licenseFileName;
1184     }

1186     /**
1187      * Download the licenses associated with this project
1188      *
1189      * @param depProject The project which generated the dependency
1190      * @param matchingUrlsOnly
1191      * @throws MojoFailureException
1192      */
1193     private void downloadLicenses( LicenseDownloader licenseDownloader, ProjectLicenseInfo depProject,
1194                                    boolean matchingUrlsOnly )
1195         throws MojoFailureException
1196     {
1197         LOG.debug( "Downloading license(s) for project {}", depProject );
1198 
1199         List<ProjectLicense> licenses = depProject.getLicenses();
1200 
1201         if ( matchingUrlsOnly && ( depProject.getLicenses() == null || depProject.getLicenses().isEmpty() ) )
1202         {
1203             handleError( depProject, "No license information available for: " + depProject.toGavString() );
1204             return;
1205         }
1206 
1207         int licenseIndex = 0;
1208         for ( ProjectLicense license : licenses )
1209         {
1210             if ( matchingUrlsOnly && license.getUrl() == null )
1211             {
1212                 handleError( depProject, "No URL for license at index " + licenseIndex + " in dependency "
1213                     + depProject.toGavString() );
1214             }
1215             else if ( license.getUrl() != null )
1216             {
1217                 final String licenseUrl = urlReplacements.rewriteIfNecessary( license.getUrl() );
1218 
1219                 final LicenseDownloadResult cachedResult = cache.get( licenseUrl );
1220                 try
1221                 {
1222 
1223                     if ( cachedResult != null )
1224                     {
1225                         if ( cachedResult.isPreferredFileName() == matchingUrlsOnly )
1226                         {
1227                             if ( organizeLicensesByDependencies )
1228                             {
1229                                 final FileNameEntry fileNameEntry =
1230                                     getLicenseFileName( depProject, licenseUrl, license.getName(), license.getFile() );
1231                                 final File cachedFile = cachedResult.getFile();
1232                                 final LicenseDownloadResult byDepsResult;
1233                                 final File byDepsFile = fileNameEntry.getFile();
1234                                 if ( cachedResult.isSuccess() && !cachedFile.equals( byDepsFile ) )
1235                                 {
1236                                     Files.copy( cachedFile.toPath(), byDepsFile.toPath() );
1237                                     byDepsResult = cachedResult.withFile( byDepsFile );
1238                                 }
1239                                 else
1240                                 {
1241                                     byDepsResult = cachedResult;
1242                                 }
1243                                 handleResult( licenseUrl, byDepsResult, depProject, license );
1244                             }
1245                             else
1246                             {
1247                                 handleResult( licenseUrl, cachedResult, depProject, license );
1248                             }
1249                         }
1250                     }
1251                     else
1252                     {
1253                         /* No cache entry for the current URL */
1254                         final FileNameEntry fileNameEntry =
1255                             getLicenseFileName( depProject, licenseUrl, license.getName(), license.getFile() );
1256 
1257                         final File licenseOutputFile = fileNameEntry.getFile();
1258                         if ( matchingUrlsOnly == fileNameEntry.isPreferred() )
1259                         {
1260                             if ( !licenseOutputFile.exists() || forceDownload )
1261                             {
1262                                 LicenseDownloadResult result =
1263                                     licenseDownloader.downloadLicense( licenseUrl, fileNameEntry );
1264                                 if ( !organizeLicensesByDependencies && result.isSuccess() )
1265                                 {
1266                                     /* check if we can re-use an existing file that has the same content */
1267                                     final String name = preferredFileNames.getFileNameBySha1( result.getSha1() );
1268                                     if ( name != null )
1269                                     {
1270                                         final File oldFile = result.getFile();
1271                                         if ( !oldFile.getName().equals( name ) )
1272                                         {
1273                                             LOG.debug(
1274                                                 "Found preferred name '{}' by SHA1 after downloading '{}'; "
1275                                                 + "renaming from '{}'",
1276                                                 name,
1277                                                 licenseUrl,
1278                                                 oldFile.getName() );
1279                                             final File newFile = new File( licensesOutputDirectory, name );
1280                                             if ( newFile.exists() )
1281                                             {
1282                                                 oldFile.delete();
1283                                             }
1284                                             else
1285                                             {
1286                                                 oldFile.renameTo( newFile );
1287                                             }
1288                                             result = result.withFile( newFile );
1289                                         }
1290                                     }
1291                                 }
1292                                 handleResult( licenseUrl, result, depProject, license );
1293                                 cache.put( licenseUrl, result );
1294                             }
1295                             else if ( licenseOutputFile.exists() )
1296                             {
1297                                 final LicenseDownloadResult result =
1298                                     LicenseDownloadResult.success( licenseOutputFile,
1299                                                                    FileUtil.sha1( licenseOutputFile.toPath() ),
1300                                                                    fileNameEntry.isPreferred() );
1301                                 handleResult( licenseUrl, result, depProject, license );
1302                                 cache.put( licenseUrl, result );
1303                             }
1304                         }
1305                     }
1306                 }
1307                 catch ( URISyntaxException e )
1308                 {
1309                     String msg = "POM for dependency " + depProject.toGavString()
1310                         + " has an invalid license URL: " + licenseUrl;
1311                     handleError( depProject, msg );
1312                     LOG.debug( msg, e );
1313                 }
1314                 catch ( FileNotFoundException e )
1315                 {
1316                     String msg = "POM for dependency " + depProject.toGavString()
1317                         + " has a license URL that returns file not found: " + licenseUrl;
1318                     handleError( depProject, msg );
1319                     LOG.debug( msg, e );
1320                 }
1321                 catch ( IOException e )
1322                 {
1323                     String msg = "Unable to retrieve license from URL '" + licenseUrl + "' for dependency '"
1324                         + depProject.toGavString() + "': " + e.getMessage();
1325                     handleError( depProject, msg );
1326                     LOG.debug( msg, e );
1327                 }
1328             }
1329             licenseIndex++;
1330         }

1348     private void handleError( ProjectLicenseInfo depProject, String msg ) throws MojoFailureException
1349     {
1350         if ( depProject.isApproved() )
1351         {
1352             LOG.debug( "Supressing manually approved license issue: {}", msg );
1353         }
1354         else
1355         {
1356             switch ( errorRemedy )
1357             {
1358                 case ignore:
1359                     /* do nothing */
1360                     break;
1361                 case warn:
1362                     LOG.warn( msg );
1363                     break;
1364                 case failFast:
1365                     throw new MojoFailureException( msg );
1366                 case xmlOutput:
1367                     LOG.error( msg );
1368                     depProject.addDownloaderMessage( msg );
1369                     break;
1370                 default:
1371                     throw new IllegalStateException( "Unexpected value of " + ErrorRemedy.class.getName() + ": "
1372                         + errorRemedy );
1373             }
1374             downloadErrorCount++;
1375         }
