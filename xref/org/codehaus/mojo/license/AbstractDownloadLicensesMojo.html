102     /**
103      * A file containing the license data (most notably license names and license URLs) missing in
104      * {@code pom.xml} files of the dependencies.
105      * <p>
106      * Note that since 1.18, if you set {@link #errorRemedy} to {@code xmlOutput} the format of
107      * {@link #licensesErrorsFile} is the same as the one of {@link #licensesConfigFile}. So you can use
108      * {@link #licensesErrorsFile} as a base for {@link #licensesConfigFile}.
109      * <p>
110      * Since 1.18, the format of the file is as follows:
111      * <pre>
112      * {@code
113      * <licenseSummary>
114      *   <dependencies>
115      *     <dependency>
116      *       <groupId>\Qaopalliance\E</groupId><!-- A regular expression -->
117      *       <artifactId>\Qaopalliance\E</artifactId><!-- A regular expression -->
118      *       <!-- <version>.*</version> A version pattern is optional, .* being the default.
119      *       <matchLicenses>
120      *         <!-- Match a list of licenses with a single entry having name "Public Domain" -->
121      *         <license>
122      *           <name>\QPublic Domain\E</name><!-- A regular expression -->
123      *         </license>
124      *       </matchLicenses>
125      *       <licenses approved="true" /><!-- Leave the matched dependency as is. -->
126      *                                   <!-- In this particular case we approve that code in the Public -->
127      *                                   <!-- Domain does not need any license URL. -->
128      *     </dependency>
129      *     <dependency>
130      *         <groupId>\Qasm\E</groupId>
131      *       <artifactId>\Qasm\E</artifactId>
132      *       <matchLicenses>
133      *         <!-- Match an empty list of licenses -->
134      *       </matchLicenses>
135      *       <!-- Replace the list of licenses in all matching dependencies with the following licenses -->
136      *       <licenses>
137      *         <license>
138      *           <name>BSD 3-Clause ASM</name>
139      *           <url>https://gitlab.ow2.org/asm/asm/raw/ASM_3_1_MVN/LICENSE.txt</url>
140      *         </license>
141      *       </licenses>
142      *     </dependency>
143      *     <dependency>
144      *       <groupId>\Qca.uhn.hapi\E</groupId>
145      *       <artifactId>.*</artifactId>
146      *       <matchLicenses>
147      *         <!-- Match a list of licenses with the following three entries in order: -->
148      *         <license>
149      *           <name>\QHAPI is dual licensed (MPL, GPL)\E</name>
150      *           <comments>\QHAPI is dual licensed under both the Mozilla Public License and the GNU General Public License.\E\s+\QWhat this means is that you may choose to use HAPI under the terms of either license.\E\s+\QYou are both permitted and encouraged to use HAPI, royalty-free, within your applications,\E\s+\Qwhether they are free/open-source or commercial/closed-source, provided you abide by the\E\s+\Qterms of one of the licenses below.\E\s+\QYou are under no obligations to inform the HAPI project about what you are doing with\E\s+\QHAPI, but we would love to hear about it anyway!\E</comments>
151      *         </license>
152      *         <license>
153      *           <name>\QMozilla Public License 1.1\E</name>
154      *           <url>\Qhttp://www.mozilla.org/MPL/MPL-1.1.txt\E</url>
155      *           <file>\Qmozilla public license 1.1 - index.0c5913925d40.txt\E</file>
156      *         </license>
157      *         <license>
158      *           <name>\QGNU General Public License\E</name>
159      *           <url>\Qhttp://www.gnu.org/licenses/gpl.txt\E</url>
160      *           <file>\Qgnu general public license - gpl.txt\E</file>
161      *         </license>
162      *       </matchLicenses>
163      *       <licenses approved="true" /><!-- It is OK that the first entry has no URL -->
164      *     </dependency>
165      *   </dependencies>
166      * </licenseSummary>
167      * }
168      * </pre>
169      * <p>
170      * Before 1.18 the format was the same as the one of {@link #licensesOutputFile} and the groupIds and artifactIds
171      * were matched as plain text rather than regular expressions. No other elements (incl. versions) were matched at
172      * all. Since 1.18 the backwards compatibility is achieved by falling back to plain text matching of groupIds
173      * and artifactIds if the given {@code <dependency>} does not contain the {@code <matchLicenses>} element.
174      * <p>
175      * Relationship to other parameters:
176      * <ul>
177      * <li>License names and license URLs {@link #licensesConfigFile} is applied before
178      * {@link #licenseUrlReplacements}</li>
179      * <li>{@link #licenseUrlReplacements} are applied before {@link #licenseUrlFileNames}</li>
180      * <li>{@link #licenseUrlFileNames} have higher precedence than {@code <file>} elements in
181      * {@link #licensesConfigFile}</li>
182      * <li>{@link #licenseUrlFileNames} are ignored when {@link #organizeLicensesByDependencies} is {@code true}</li>
183      * </ul>
184      *
185      * @since 1.0
186      */
187     @Parameter( property = "licensesConfigFile", defaultValue = "${project.basedir}/src/license/licenses.xml" )
188     protected File licensesConfigFile;

448      * List of regexps/replacements applied to the license urls prior to download.

454      *
455      * <p>If the replacement element is omitted, this is equivalent to an empty replacement string.</p>
456      *
457      * <pre>
458      * {@code
459      *
460      * <licenseUrlReplacements>
461      *   <licenseUrlReplacement>
462      *     <regexp>\Qhttps://glassfish.java.net/public/CDDL+GPL_1_1.html\E</regexp>
463      *     <replacement>https://oss.oracle.com/licenses/CDDL+GPL-1.1</replacement>
464      *   </licenseUrlReplacement>
465      *   <licenseUrlReplacement>
466      *     <regexp>https://(.*)</regexp><!-- replace https with http -->
467      *     <replacement>http://$1</replacement>;
468      *   </licenseUrlReplacement>
469      *   <licenseUrlReplacement>
470      *     <regexp>^https?://github\.com/([^/]+)/([^/]+)/blob/(.*)$</regexp><!-- replace GitHub web UI with raw -->
471      *     <replacement>https://raw.githubusercontent.com/$1/$2/$3</replacement>
472      *   </licenseUrlReplacement>
473      * </licenseUrlReplacements>
474      * }
475      * </pre>
476      * <p>
477      * Relationship to other parameters:
478      * <ul>
479      * <li>License names and license URLs {@link #licensesConfigFile} is applied before
480      * {@link #licenseUrlReplacements}</li>
481      * <li>{@link #licenseUrlReplacements} are applied before {@link #licenseUrlFileNames}</li>
482      * <li>{@link #licenseUrlFileNames} have higher precedence than {@code <file>} elements in
483      * {@link #licensesConfigFile}</li>
484      * <li>{@link #licenseUrlFileNames} are ignored when {@link #organizeLicensesByDependencies} is {@code true}</li>

492     /**
493      * A map that helps to select local files names for the content downloaded from license URLs.
494      * <p>
495      * Keys in the map are the local file names. These files will be created under {@link #licensesOutputDirectory}.
496      * <p>
497      * Values are white space ({@code " \t\n\r"}) separated lists of regular expressions that will be used to match
498      * license URLs. The regular expressions are compiled using {@link Pattern#CASE_INSENSITIVE}. Note that various
499      * characters that commonly occur in URLs have special meanings in regular extensions. Therefore, consider using
500      * regex quoting as described in {@link Pattern} - e.g. {@code http://example\.com} or
501      * {@code \Qhttp://example.com\E}
502      * <p>
503      * In addition to URL patterns, the list can optionally contain a sha1 checksum of the expected content. This is to
504      * ensure that the content delivered by a URL does not change without notice. Note that strict checking
505      * of the checksum happens only when {@link #forceDownload} is {@code true}. Otherwise the mojo assumes that the URL
506      * -&gt; local name mapping is correct and downloads from the URL only if the local file does not exist.
507      * <p>
508      * A special value-less entry {@code <spdx/>} can be used to activate built-in license names that are based on
509      * license IDs from <a href="https://spdx.org/licenses/">https://spdx.org/licenses</a>. The built-in SPDX mappings
510      * can be overridden by the subsequent entries. To see which SPDX mappings are built-in, add the {@code <spdx/>}
511      * entry and run the mojo with debug log level, e.g. using {@code -X} or
512      * {-Dorg.slf4j.simpleLogger.log.org.codehaus.mojo.license=debug} on the command line.
513      * <p>
514      * An example:
515      * <pre>
516      * {@code
517      * <licenseUrlFileNames>
518      *   <spdx/><!-- A special element to activate built-in file name entries based on spdx.org license IDs -->
519      *   <bsd-antlr.html>
520      *       sha1:81ffbd1712afe8cdf138b570c0fc9934742c33c1
521      *       https?://(www\.)?antlr\.org/license\.html
522      *   </bsd-antlr.html>
523      *   <cddl-gplv2-ce.txt>
524      *       sha1:534a3fc9ae1076409bb00d01127dbba1e2620e92
525      *       \Qhttps://raw.githubusercontent.com/javaee/activation/master/LICENSE.txt\E
526      *   </cddl-gplv2-ce.txt>
527      * </licenseUrlFileNames>
528      * }
529      * </pre>
530      * <p>
531      * Relationship to other parameters:
532      * <ul>
533      * <li>License names and license URLs {@link #licensesConfigFile} is applied before
534      * {@link #licenseUrlReplacements}</li>
535      * <li>{@link #licenseUrlReplacements} are applied before {@link #licenseUrlFileNames}</li>
536      * <li>{@link #licenseUrlFileNames} have higher precedence than {@code <file>} elements in
537      * {@link #licensesConfigFile}</li>
538      * <li>{@link #licenseUrlFileNames} are ignored when {@link #organizeLicensesByDependencies} is {@code true}</li>
539      * </ul>
540      *
541      * @since 1.18
542      */
543     @Parameter
544     protected Map<String, String> licenseUrlFileNames;

639     /**
640      * {@inheritDoc}
641      * @throws MojoFailureException
642      */
643     public void execute()
644         throws MojoExecutionException, MojoFailureException
645     {
646 
647         if ( isSkip() )
648         {
649             getLog().info( "skip flag is on, will skip goal." );
650             return;
651         }
652 
653         this.errorRemedy = getEffectiveErrorRemedy( this.quiet, this.errorRemedy );
654         this.preferredFileNames = PreferredFileNames.build( licensesOutputDirectory, licenseUrlFileNames, getLog() );
655         this.cache = new Cache( licenseUrlFileNames != null && !licenseUrlFileNames.isEmpty() );
656 
657         initDirectories();
658 
659         if ( licensesOutputFileEncoding == null )
660         {
661             licensesOutputFileEncoding = System.getProperty( "file.encoding" );
662             getLog().warn( "Using the default system encoding for reading or writing licenses.xml file."
663                     + " This makes your build platform dependent. You should set either"
664                     + " project.build.sourceEncoding or licensesOutputFileEncoding" );
665         }
666         final Charset charset = Charset.forName( licensesOutputFileEncoding );
667 
668         final LicenseMatchers matchers = LicenseMatchers.load( licensesConfigFile );
669 
670         final Set<MavenProject> dependencies = getDependencies();
671 
672         // The resulting list of licenses after dependency resolution
673         final List<ProjectLicenseInfo> depProjectLicenses = new ArrayList<>();
674 
675         try ( LicenseDownloader licenseDownloader =
676             new LicenseDownloader( findActiveProxy(), connectTimeout, socketTimeout, connectionRequestTimeout ) )
677         {
678             for ( MavenProject project : dependencies )
679             {
680                 Artifact artifact = project.getArtifact();
681                 getLog().debug( "Checking licenses for project " + artifact );
682                 final ProjectLicenseInfo depProject = createDependencyProject( project );
683                 matchers.replaceMatches( depProject );
684                 depProjectLicenses.add( depProject );
685             }
686             if ( !offline )
687             {
688                 /* First save the matching URLs into the cache */
689                 for ( ProjectLicenseInfo depProject : depProjectLicenses )
690                 {
691                     downloadLicenses( licenseDownloader, depProject, true );
692                 }
693                 getLog().debug( "Finished populating cache" );
694                 /*
695                  * Then attempt to download the rest of the URLs using the available cache entries to select local
696                  * file names based on file content sha1
697                  */
698                 for ( ProjectLicenseInfo depProject : depProjectLicenses )
699                 {
700                     downloadLicenses( licenseDownloader, depProject, false );
701                 }
702             }
703         }
704         catch ( IOException e )
705         {
706             throw new RuntimeException( e );
707         }
708 
709         try
710         {
711             if ( sortByGroupIdAndArtifactId )
712             {
713                 sortByGroupIdAndArtifactId( depProjectLicenses );
714             }
715 
716             if ( licensesOutputFileEol == Eol.AUTODETECT )
717             {
718                 final Path autodetectFromFile = licensesConfigFile.exists() ? licensesConfigFile.toPath()
719                         : project.getBasedir().toPath().resolve( "pom.xml" );
720                 licensesOutputFileEol = Eol.autodetect( autodetectFromFile, charset );
721             }
722 
723             List<ProjectLicenseInfo> depProjectLicensesWithErrors = filterErrors( depProjectLicenses );
724             LicenseSummaryWriter.writeLicenseSummary( depProjectLicenses, licensesOutputFile, charset,
725                                                       licensesOutputFileEol, writeVersions );
726             if ( depProjectLicensesWithErrors != null && !depProjectLicensesWithErrors.isEmpty() )
727             {
728                 LicenseSummaryWriter.writeLicenseSummary( depProjectLicensesWithErrors, licensesErrorsFile, charset,
729                                                           licensesOutputFileEol, writeVersions );
730             }
731         }
732         catch ( Exception e )
733         {
734             throw new MojoExecutionException( "Unable to write license summary file: " + licensesOutputFile, e );
735         }
736 
737         switch ( errorRemedy )
738         {
739             case ignore:
740             case failFast:
741                 /* do nothing */
742                 break;
743             case warn:
744                 getLog().warn( "There were " + downloadErrorCount + " download errors - check the warnings above" );
745                 break;
746             case xmlOutput:
747                 if ( downloadErrorCount > 0 )
748                 {
749                     throw new MojoFailureException( "There were " + downloadErrorCount + " download errors - check "
750                         + licensesErrorsFile.getAbsolutePath() );
751                 }
752                 break;
753             default:
754                 throw new IllegalStateException( "Unexpected value of " + ErrorRemedy.class.getName() + ": "
755                     + errorRemedy );
756         }
757     }

759     /**
760      * Removes from the given {@code depProjectLicenses} those elements which have non-empty
761      * {@link ProjectLicenseInfo#getDownloaderMessages()} and adds those to the resulting {@link List}.
762      *
763      * @param depProjectLicenses the list of {@link ProjectLicenseInfo}s to filter
764      * @return a new {@link List} of {@link ProjectLicenseInfo}s containing only elements with non-empty
765      *         {@link ProjectLicenseInfo#getDownloaderMessages()}
766      */
767     private List<ProjectLicenseInfo> filterErrors( List<ProjectLicenseInfo> depProjectLicenses )
768     {
769         final List<ProjectLicenseInfo> result = new ArrayList<>();
770         final Iterator<ProjectLicenseInfo> it = depProjectLicenses.iterator();
771         while ( it.hasNext() )
772         {
773             final ProjectLicenseInfo dep = it.next();
774             final List<String> messages = dep.getDownloaderMessages();
775             if ( messages != null && !messages.isEmpty() )
776             {
777                 it.remove();
778                 result.add( dep );
779             }
780         }
781         return result;
782     }

784     private static ErrorRemedy getEffectiveErrorRemedy( boolean quiet, ErrorRemedy errorRemedy )
785     {
786         switch ( errorRemedy )
787         {
788             case warn:
789                 return quiet ? ErrorRemedy.ignore : ErrorRemedy.warn;
790             default:
791                 return errorRemedy;
792         }

795     private void sortByGroupIdAndArtifactId( List<ProjectLicenseInfo> depProjectLicenses )
796     {
797         Comparator<ProjectLicenseInfo> comparator = new Comparator<ProjectLicenseInfo>()
798         {
799             public int compare( ProjectLicenseInfo info1, ProjectLicenseInfo info2 )
800             {
801                 //ProjectLicenseInfo::getId() can not be used because . is before : thus a:b.c would be after a.b:c
802                 return ( info1.getGroupId() + "+" + info1.getArtifactId() ).compareTo( info2.getGroupId()
803                         + "+" + info2.getArtifactId() );
804             }
805         };
806         Collections.sort( depProjectLicenses, comparator );
807     }

991     /**
992      * Determine filename to use for downloaded license file. The file name is based on the configured name of the
993      * license (if available) and the remote filename of the license.
994      *
995      * @param depProject the project containing the license
996      * @param licenseUrl the license url
997      * @param licenseName the license name
998      * @param string
999      * @return A filename to be used for the downloaded license file
1000      * @throws URISyntaxException
1001      */
1002     private FileNameEntry getLicenseFileName( ProjectLicenseInfo depProject, final String url,
1003                                                            final String licenseName, String licenseFileName )
1004         throws URISyntaxException
1005     {
1006 
1007         final URI licenseUrl = new URI( url );
1008         File licenseUrlFile = new File( licenseUrl.getPath() );
1009 
1010         if ( organizeLicensesByDependencies )
1011         {
1012             if ( licenseFileName != null && !licenseFileName.isEmpty() )
1013             {
1014                 return new FileNameEntry( new File( licensesOutputDirectory, new File( licenseFileName ).getName() ),
1015                                           false, null );
1016             }
1017             licenseFileName = String.format( "%s.%s%s", depProject.getGroupId(), depProject.getArtifactId(),
1018                                              licenseName != null
1019                                                  ? "_" + licenseName
1020                                                  : "" ).replaceAll( "\\s+", "_" );
1021         }
1022         else
1023         {
1024             final FileNameEntry preferredFileNameEntry = preferredFileNames.getEntryByUrl( url );
1025             if ( preferredFileNameEntry != null )
1026             {
1027                 return preferredFileNameEntry;
1028             }
1029 
1030             if ( licenseFileName != null && !licenseFileName.isEmpty() )
1031             {
1032                 return new FileNameEntry( new File( licensesOutputDirectory,
1033                                                                          new File( licenseFileName ).getName() ),
1034                                                                false, null );
1035             }
1036 
1037             licenseFileName = licenseUrlFile.getName();
1038 
1039             if ( licenseName != null )
1040             {
1041                 licenseFileName = licenseName + " - " + licenseUrlFile.getName();
1042             }
1043 
1044             // Normalize whitespace
1045             licenseFileName = licenseFileName.replaceAll( "\\s+", " " );
1046         }
1047 
1048         // lower case and (back)slash removal
1049         licenseFileName = licenseFileName.toLowerCase( Locale.US ).replaceAll( "[\\\\/]+", "_" );
1050 
1051         licenseFileName = sanitize( licenseFileName );
1052 
1053         return new FileNameEntry( new File( licensesOutputDirectory, licenseFileName ), false, null );
1054     }
1055 
1056     private String sanitize( String licenseFileName )
1057     {
1058         if ( licenseUrlFileNameSanitizers != null )
1059         {
1060             for ( LicenseUrlReplacement sanitizer : licenseUrlFileNameSanitizers )
1061             {
1062                 Pattern regexp = sanitizer.getRegexp();
1063                 String replacement = sanitizer.getReplacement() == null ? "" : sanitizer.getReplacement();
1064                 if ( regexp != null )
1065                 {
1066                     licenseFileName = regexp.matcher( licenseFileName ).replaceAll( replacement );
1067                 }
1068             }
1069         }
1070         return licenseFileName;
1071     }

1073     /**
1074      * Download the licenses associated with this project
1075      *
1076      * @param depProject The project which generated the dependency
1077      * @param matchingUrlsOnly
1078      * @throws MojoFailureException
1079      */
1080     private void downloadLicenses( LicenseDownloader licenseDownloader, ProjectLicenseInfo depProject,
1081                                    boolean matchingUrlsOnly )
1082         throws MojoFailureException
1083     {
1084         getLog().debug( "Downloading license(s) for project " + depProject );
1085 
1086         List<ProjectLicense> licenses = depProject.getLicenses();
1087 
1088         if ( matchingUrlsOnly && ( depProject.getLicenses() == null || depProject.getLicenses().isEmpty() ) )
1089         {
1090             handleError( depProject, "No license information available for: " + depProject );
1091             return;
1092         }
1093 
1094         int licenseIndex = 0;
1095         for ( ProjectLicense license : licenses )
1096         {
1097             if ( matchingUrlsOnly && license.getUrl() == null )
1098             {
1099                 handleError( depProject, "No URL for license at index " + licenseIndex + " in dependency "
1100                     + depProject.toString() );
1101             }
1102             else if ( license.getUrl() != null )
1103             {
1104                 final String licenseUrl = rewriteLicenseUrlIfNecessary( license.getUrl() );
1105 
1106                 final LicenseDownloadResult cachedResult = cache.get( licenseUrl );
1107                 try
1108                 {
1109 
1110                     if ( cachedResult != null )
1111                     {
1112                         if ( cachedResult.isPreferredFileName() == matchingUrlsOnly )
1113                         {
1114                             if ( organizeLicensesByDependencies )
1115                             {
1116                                 final FileNameEntry fileNameEntry =
1117                                     getLicenseFileName( depProject, licenseUrl, license.getName(), license.getFile() );
1118                                 final File cachedFile = cachedResult.getFile();
1119                                 final LicenseDownloadResult byDepsResult;
1120                                 final File byDepsFile = fileNameEntry.getFile();
1121                                 if ( cachedResult.isSuccess() && !cachedFile.equals( byDepsFile ) )
1122                                 {
1123                                     Files.copy( cachedFile.toPath(), byDepsFile.toPath() );
1124                                     byDepsResult = cachedResult.withFile( byDepsFile );
1125                                 }
1126                                 else
1127                                 {
1128                                     byDepsResult = cachedResult;
1129                                 }
1130                                 handleResult( licenseUrl, byDepsResult, depProject, license );
1131                             }
1132                             else
1133                             {
1134                                 handleResult( licenseUrl, cachedResult, depProject, license );
1135                             }
1136                         }
1137                     }
1138                     else
1139                     {
1140                         /* No cache entry for the current URL */
1141                         final FileNameEntry fileNameEntry =
1142                             getLicenseFileName( depProject, licenseUrl, license.getName(), license.getFile() );
1143 
1144                         final File licenseOutputFile = fileNameEntry.getFile();
1145                         if ( matchingUrlsOnly == fileNameEntry.isPreferred() )
1146                         {
1147                             if ( !licenseOutputFile.exists() || forceDownload )
1148                             {
1149                                 LicenseDownloadResult result =
1150                                     licenseDownloader.downloadLicense( licenseUrl, fileNameEntry, getLog() );
1151                                 if ( !organizeLicensesByDependencies && result.isSuccess() )
1152                                 {
1153                                     /* check if we can re-use an existing file that has the same content */
1154                                     final String name = preferredFileNames.getFileNameBySha1( result.getSha1() );
1155                                     if ( name != null )
1156                                     {
1157                                         final File oldFile = result.getFile();
1158                                         if ( !oldFile.getName().equals( name ) )
1159                                         {
1160                                             getLog().debug( "Found preferred name '" + name
1161                                                 + "' by sha1 after downloading '" + licenseUrl + "'; renaming from '"
1162                                                 + oldFile.getName() + "'" );
1163                                             final File newFile = new File( licensesOutputDirectory, name );
1164                                             if ( newFile.exists() )
1165                                             {
1166                                                 oldFile.delete();
1167                                             }
1168                                             else
1169                                             {
1170                                                 oldFile.renameTo( newFile );
1171                                             }
1172                                             result = result.withFile( newFile );
1173                                         }
1174                                     }
1175                                 }
1176                                 handleResult( licenseUrl, result, depProject, license );
1177                                 cache.put( licenseUrl, result );
1178                             }
1179                             else if ( licenseOutputFile.exists() )
1180                             {
1181                                 final LicenseDownloadResult result =
1182                                     LicenseDownloadResult.success( licenseOutputFile,
1183                                                                    FileUtil.sha1( licenseOutputFile.toPath() ),
1184                                                                    fileNameEntry.isPreferred() );
1185                                 handleResult( licenseUrl, result, depProject, license );
1186                                 cache.put( licenseUrl, result );
1187                             }
1188                         }
1189                     }
1190                 }
1191                 catch ( URISyntaxException e )
1192                 {
1193                     handleError( depProject, "POM for dependency " + depProject.toString()
1194                         + " has an invalid license URL: " + licenseUrl );
1195                     getLog().debug( e );
1196                 }
1197                 catch ( FileNotFoundException e )
1198                 {
1199                     handleError( depProject, "POM for dependency " + depProject.toString()
1200                         + " has a license URL that returns file not found: " + licenseUrl );
1201                     getLog().debug( e );
1202                 }
1203                 catch ( IOException e )
1204                 {
1205                     handleError( depProject, "Unable to retrieve license from URL '" + licenseUrl + "' for dependency '"
1206                         + depProject.toString() + "': " + e.getMessage() );
1207                     getLog().debug( e );
1208                 }
1209             }
1210             licenseIndex++;
1211         }
1212 
1213     }

1215     private void handleResult( String licenseUrl, LicenseDownloadResult result, ProjectLicenseInfo depProject,
1216                                ProjectLicense license )
1217         throws MojoFailureException
1218     {
1219         if ( result.isSuccess() )
1220         {
1221             license.setFile( result.getFile().getName() );
1222         }
1223         else

1229     private void handleError( ProjectLicenseInfo depProject, String msg ) throws MojoFailureException
1230     {
1231         if ( depProject.isApproved() )
1232         {
1233             getLog().debug( "Supressing manually approved license issue: " + msg );
1234         }
1235         else
1236         {
1237             switch ( errorRemedy )
1238             {
1239                 case ignore:
1240                     /* do nothing */
1241                     break;
1242                 case warn:
1243                     getLog().warn( msg );
1244                     break;
1245                 case failFast:
1246                     throw new MojoFailureException( msg );
1247                 case xmlOutput:
1248                     getLog().debug( msg );
1249                     depProject.addDownloaderMessage( msg );
1250                     break;
1251                 default:
1252                     throw new IllegalStateException( "Unexpected value of " + ErrorRemedy.class.getName() + ": "
1253                         + errorRemedy );
1254             }
1255             downloadErrorCount++;
1256         }
1257     }

1259     private String rewriteLicenseUrlIfNecessary( final String originalLicenseUrl )
1260     {
1261         String resultUrl = originalLicenseUrl;
1262         if ( licenseUrlReplacements != null )
1263         {
1264             for ( LicenseUrlReplacement urlReplacement : licenseUrlReplacements )
1265             {
1266                 Pattern regexp = urlReplacement.getRegexp();
1267                 String replacement = urlReplacement.getReplacement() == null ? "" : urlReplacement.getReplacement();
1268                 if ( regexp != null )
1269                 {
1270                     resultUrl = regexp.matcher( resultUrl ).replaceAll( replacement );
1271                 }
1272             }
1273 
1274             if ( !resultUrl.equals( originalLicenseUrl ) )
1275             {
1276                 getLog().debug( String.format( "Rewrote URL %s => %s", originalLicenseUrl, resultUrl ) );
1277             }
1278         }
1279         return resultUrl;
1280     }
