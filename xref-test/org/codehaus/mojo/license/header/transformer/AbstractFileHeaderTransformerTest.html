107     @Test
108     public void testCopyrightPattern()
109         throws Exception
110     {
111         String actual;
112         Matcher matcher;
113         String prefix;
114         String firstYear;
115         String lastYear;
116         String holder;
117 
118         actual = "Copyright (C) 2010 Tony";
119         matcher = COPYRIGHT_PATTERN.matcher( actual );
120         Assert.assertTrue( matcher.matches() );
121         Assert.assertEquals( 5, matcher.groupCount() );
122         prefix = matcher.group( 1 );
123         Assert.assertEquals( "Copyright (C)", prefix );
124         firstYear = matcher.group( 2 );
125         Assert.assertEquals( "2010", firstYear );
126         lastYear = matcher.group( 4 );
127         Assert.assertEquals( null, lastYear );
128         holder = matcher.group( 5 );
129         Assert.assertEquals( "Tony", holder );
130 
131         actual = "Copyright (C) 2010 - 2012    Tony";
132         matcher = COPYRIGHT_PATTERN.matcher( actual );
133         Assert.assertTrue( matcher.matches() );
134         Assert.assertEquals( 5, matcher.groupCount() );
135         prefix = matcher.group( 1 );
136         Assert.assertEquals( "Copyright (C)", prefix );
137         firstYear = matcher.group( 2 );
138         Assert.assertEquals( "2010", firstYear );
139         lastYear = matcher.group( 4 );
140         Assert.assertEquals( "2012", lastYear );
141         holder = matcher.group( 5 );
142         Assert.assertEquals( "Tony", holder );
143 
144         // Test MLICENSE-34 (multi-line copyright holders)
145         actual = "Copyright (C) 2010 - 2012    Tony\nAnd Yet\n Another\nLine";
146         matcher = COPYRIGHT_PATTERN.matcher( actual );
147         Assert.assertTrue( matcher.matches() );
148         Assert.assertEquals( 5, matcher.groupCount() );
149         prefix = matcher.group( 1 );
150         Assert.assertEquals( "Copyright (C)", prefix );
151         firstYear = matcher.group( 2 );
152         Assert.assertEquals( "2010", firstYear );
153         lastYear = matcher.group( 4 );
154         Assert.assertEquals( "2012", lastYear );
155         holder = matcher.group( 5 );
156         Assert.assertEquals( "Tony\nAnd Yet\n Another\nLine", holder );
157     }

245     {
246         String header;

248

259                                  commentLinePrefix + "License" + LINE_SEPARATOR + commentEndTag + LINE_SEPARATOR,

265             commentLinePrefix + "Description" + LINE_SEPARATOR + commentLinePrefix + DEFAULT_SECTION_DELIMITER +
266                 LINE_SEPARATOR + commentLinePrefix + "Copyright (C) 2010 Tony" + LINE_SEPARATOR +
267                 commentLinePrefix + DEFAULT_SECTION_DELIMITER + LINE_SEPARATOR + commentLinePrefix + "License" +
